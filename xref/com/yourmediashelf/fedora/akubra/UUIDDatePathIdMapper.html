108 	 *
109 	 * @param dateFormat A {@link DateTimeFormat} pattern to generate the path.
110      * If null or empty, defaults to "yyyy/MM/dd".
111 	 * @param fallbackMapper The {@link IdMapper} to use if/when we encounter an
112      * id that is not a version 1 UUID. If null, defaults to TrivialIdMapper.
113 	 * @param prefixer The IdMapperPrefixer to use, or <code>null</code>.
114 	 *
115 	 * @see <a href="http://joda-time.sourceforge.net/apidocs/org/joda/time/format/DateTimeFormat.html#forPattern%28java.lang.String%29>Joda-Time DateTimeFormat</a>
116 	 */
117 	public UUIDDatePathIdMapper(String dateFormat, IdMapper fallbackMapper, IdMapperPrefixer prefixer) {
118         // minor performance optimization since we only support UTC
119         System.setProperty("org.joda.time.DateTimeZone.Provider", "org.joda.time.tz.UTCProvider");
120 
121         // regex Pattern for canonical UUIDs (32 hexadecimal digits, separated in 5 groups by hyphens)
122         pattern = Pattern.compile(".*([0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}).*");
123 
124         if (dateFormat == null || dateFormat.isEmpty()) {
125             dateFormat = "yyyy/MM/dd";
126         }
127         fmt = DateTimeFormat.forPattern(dateFormat);
128 
129         if (fallbackMapper == null) {
130             this.fallbackMapper = new TrivialIdMapper();
131         } else {
132             this.fallbackMapper = fallbackMapper;
133         }
134 
135         this.prefixer = prefixer;
136     }

142 		String uri = internalId.toString();
143 
144 		try {
145 			extractUUID(uri);
146 		} catch (IllegalArgumentException e) {
147 			return fallbackMapper.getExternalId(internalId);
148 		}
149 		
150 		
151 		String fullPath = internalId.toString().substring(
152                 internalScheme.length() + 1);
153         int i = fullPath.lastIndexOf('/'); // e.g. everything after "info:fedora/"
154         String encodedURI;
155         if (i == -1)
156             encodedURI = fullPath;
157         else
158             encodedURI = fullPath.substring(i + 1);
159         return URI.create(IdMapperUtil.decode(encodedURI));
160 	}

191 	/**
192 	 * Given an identifier which contains a type 1 UUID in canonical form,
193 	 * return the date-formatted path given by the identifier.
194 	 *
195 	 * @param uri the identifier, e.g. "info:fedora/test:25f814ce-f5ac-11e0-b139-2837370107a5"
196 	 * @return the path for the identifier, e.g. "2011/10/13/"
197 	 * @throws NullPointerException if the uri is null
198 	 * @throws IllegalArgumentException if the uri does not contain a type 1 UUID
199 	 */
200 	private String getPath(String uri) throws NullPointerException, IllegalArgumentException {
201 		UUID uuid = extractUUID(uri);
202 		UUIDType uuidType = UUIDUtil.typeOf(uuid);
203 		if (uuidType.equals(UUIDType.TIME_BASED)) {
204 		    StringBuilder prefix = new StringBuilder();
205 		    if (prefixer != null) {
206     		    prefix.append(prefixer.getPrefix(uri));
207     		    if (prefix.length() > 0) {
208     		        prefix.append('/');
209     		    }
210 		    }
211 
212 			return prefix + getDateTime(uuid).toString(fmt) + '/';
213 
214 		} else {
215 			throw new IllegalArgumentException("Wrong type of UUID. " + uuid.toString() + " is " + uuidType);
216 		}
217 	}

219     /**
220      * Searches for and returns a UUID embedded in the supplied string.
221      * 
222      * @param s Any string that contains a UUID in canonical form.
223      * @return UUID The UUID contained in the supplied string.
224      * @throws NullPointerException
225 	 * @throws IllegalArgumentException if the input does not contain a UUID
226 	 */
227 	protected UUID extractUUID(String s) throws NullPointerException, IllegalArgumentException {
228     	if (s == null) {
229     		throw new NullPointerException();
230     	}
231     	
232     	if (s.length() < 36) {
233     		throw new IllegalArgumentException("String \"" + s + "\" is too short to be a UUID");
234     	}
235     	
236     	Matcher m = pattern.matcher(s);
237     	if (m.matches()) {
238     		return UUIDUtil.uuid(m.group(1));
239     	} else {
240     		throw new IllegalArgumentException("\"" + s + "\" is not a UUID");
241     	}
242     }

244     /**
245      * Get the DateTime from a version 1 UUID.
246      * 
247      * @param uuid a version 1 UUID
248      * @return DateTime value extracted from the UUID, or IllegalArgumentException 
249      * if uuid is not a version 1 UUID.
250      */
251     protected DateTime getDateTime(UUID uuid) {
252     	if (uuid.version() != 1) {
253     		throw new IllegalArgumentException("Wrong UUID version: " + uuid.version());
254     	}
255     	
256 		long t1 = uuid.timestamp() - NUM_100NS_INTERVALS_SINCE_UUID_EPOCH;
257     		
258 		return new DateTime(t1/10000, DateTimeZone.UTC);
259     }
