25  /**
26   * A trivial implementation of the Akubra IdMapper interface.
27   * 
28   * This IdMapper is only intended for testing or the most trivial use cases.
29   * 
30   * @author Edwin Shin
31   *
32   */
33  public class TrivialIdMapper implements IdMapper {
34  	
35  	private static final String internalScheme = "file";
36  
37  	/* (non-Javadoc)
38  	 * @see org.akubraproject.map.IdMapper#getExternalId(java.net.URI)
39  	 */
40  	public URI getExternalId(URI internalId) throws NullPointerException {
41  		String fullPath = internalId.toString().substring(
42                  internalScheme.length() + 1);
43          int i = fullPath.lastIndexOf('/');
44          String encodedURI;
45          if (i == -1)
46              encodedURI = fullPath;
47          else
48              encodedURI = fullPath.substring(i + 1);
49          return URI.create(IdMapperUtil.decode(encodedURI));
50  	}
51  
52  	/* (non-Javadoc)
53  	 * @see org.akubraproject.map.IdMapper#getInternalId(java.net.URI)
54  	 */
55  	public URI getInternalId(URI externalId) throws NullPointerException {
56  		if (externalId == null) {
57              throw new NullPointerException();
58          }
59          String uri = externalId.toString();
60          
61          return URI.create(internalScheme + ":" + IdMapperUtil.encode(uri));
62  	}
63  
64  	/* (non-Javadoc)
65  	 * @see org.akubraproject.map.IdMapper#getInternalPrefix(java.lang.String)
66  	 */
67  	public String getInternalPrefix(String externalPrefix) throws NullPointerException {
68  		if (externalPrefix == null) {
69              throw new NullPointerException();
70          }
71  		// TODO
72  		return null;
73  	}
74  }
