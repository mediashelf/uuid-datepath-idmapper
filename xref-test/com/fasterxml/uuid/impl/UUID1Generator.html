19  package com.fasterxml.uuid.impl;

21  import java.util.UUID;
22  
23  import com.fasterxml.uuid.EthernetAddress;
24  import com.fasterxml.uuid.MutableUUIDTimer;
25  import com.fasterxml.uuid.NoArgGenerator;
26  import com.fasterxml.uuid.UUIDType;
27  import com.fasterxml.uuid.impl.UUIDUtil;

29  /**
30   * UUID version 1 generator that accepts a unix timestamp as an argument.
31   * This is just a hack since some of the java-uuid-generator classes/methods 
32   * are currently final or not visible. Hopefully future versions of JUG will
33   * obviate the need for this.
34   * 
35   * @author Edwin Shin
36   *
37   */
38  public class UUID1Generator extends NoArgGenerator {
39  
40  	protected final EthernetAddress _ethernetAddress;
41  
42  	/**
43  	 * Object used for synchronizing access to timestamps, to guarantee that
44  	 * timestamps produced by this generator are unique and monotonically
45  	 * increasings. Some implementations offer even stronger guarantees, for
46  	 * example that same guarantee holds between instances running on different
47  	 * JVMs (or with native code).
48  	 */
49  	protected final MutableUUIDTimer _timer;
50  
51  	/**
52  	 * Base values for the second long (last 8 bytes) of UUID to construct
53  	 */
54  	protected final long _uuidL2;
55  
56  	public UUID1Generator(EthernetAddress ethAddr, MutableUUIDTimer timer) {
57  		/*
58  		 * UUID version 1 is a 128-bit (16 byte) value, split as: <br>
59  		 * 	32 bits time_low <br>
60  		 *  16 bits time_mid (offset 4 bytes) <br>
61  		 *  16 bits time_hi_and_version (offset 6 bytes) <br>
62  		 *  16 bits clock sequence (offset 8 bytes) <br>
63  		 *  48 bits node (offset 10 bytes)
64  		 */
65  		byte[] uuidBytes = new byte[16];
66  		if (ethAddr == null) {
67  			ethAddr = EthernetAddress.constructMulticastAddress();
68  		}
69  
70  		_ethernetAddress = ethAddr;
71  		// The node field is a 48-bit (6 byte) value
72  		_ethernetAddress.toByteArray(uuidBytes, 10);
73  
74  		// Add clock sequence
75  		int clockSeq = timer.getClockSequence();
76  		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_SEQUENCE] = (byte) (clockSeq >> 8);
77  		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_SEQUENCE + 1] = (byte) clockSeq;
78  		long l2 = UUIDUtil.gatherLong(uuidBytes, 8);
79  		_uuidL2 = UUIDUtil.initUUIDSecondLong(l2);
80  		_timer = timer;
81  	}
82  	
83  	public UUID generate(long timeInMillis) {
84  		return generateByTimestamp(_timer.getTimestamp(timeInMillis));
85  	}
86  	
87  	@Override
88  	public UUID generate() {
89  		return generateByTimestamp(_timer.getTimestamp());
90          
91  	}
92  
93  	@Override
94  	public UUIDType getType() {
95  		return UUIDType.TIME_BASED;
96  	}
97  	
98  	private UUID generateByTimestamp(long rawTimestamp) {
99  		// Time field components are kind of shuffled, need to slice:
100         int clockHi = (int) (rawTimestamp >>> 32);
101         int clockLo = (int) rawTimestamp;
102         // and dice
103         int midhi = (clockHi << 16) | (clockHi >>> 16);
104         // need to squeeze in type (4 MSBs in byte 6, clock hi)
105         midhi &= ~0xF000; // remove high nibble of 6th byte
106         midhi |= 0x1000; // type 1
107         long midhiL = (long) midhi;
108         midhiL = ((midhiL << 32) >>> 32); // to get rid of sign extension
109         // and reconstruct
110         long l1 = (((long) clockLo) << 32) | midhiL;
111         // last detail: must force 2 MSB to be '10'
112         return new UUID(l1, _uuidL2);
113 	}
