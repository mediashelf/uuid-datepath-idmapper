28  /**
29   * UUIDTimer that accepts a unix timestamp as an argument.
30   * This is just a hack since some of the java-uuid-generator classes/methods 
31   * are currently final or not visible. Hopefully future versions of JUG will
32   * obviate the need for this.
33   * 
34   * @author Edwin Shin
35   *
36   */
37  public class MutableUUIDTimer {
38  
39  	/**
40  	 * Since System.longTimeMillis() returns time from 1 January 1970, and
41  	 * UUIDs need time from the beginning of Gregorian calendar (15 October 1582),
42  	 * need to apply the offset:
43  	 */
44  	private final static long kClockOffset = 0x01b21dd213814000L;
45  	
46  	/**
47  	 * Also, instead of getting time in units of 100nsecs, we get something with
48  	 * max resolution of 1 msec... and need the multiplier as well
49  	 */
50  	private final static int kClockMultiplier = 10000;
51  
52  	private final static long kClockMultiplierL = 10000L;
53  
54  	/**
55  	 * Let's allow "virtual" system time to advance at most 100 milliseconds
56  	 * beyond actual physical system time, before adding delays.
57  	 */
58  	private final static long kMaxClockAdvance = 100L;
59  
60  	// Configuration
61  
62  	/**
63  	 * Object used to reliably ensure that no multiple JVMs generate UUIDs, and
64  	 * also that the time stamp value used for generating time-based UUIDs is
65  	 * monotonically increasing even if system clock moves backwards over a
66  	 * reboot (usually due to some system level problem).
67  	 * <p>
68  	 * See {@link TimestampSynchronizer} for details.
69  	 */
70  	protected final TimestampSynchronizer _syncer;
71  
72  	/**
73  	 * Random number generator used to generate additional information to
74  	 * further reduce probability of collisions.
75  	 */
76  	protected final Random _random;
77  
78  	// Clock state:
79  
80  	/**
81  	 * Additional state information used to protect against anomalous cases
82  	 * (clock time going backwards, node id getting mixed up). Third byte is
83  	 * actually used for seeding counter on counter overflow. Note that only
84  	 * lowermost 16 bits are actually used as sequence
85  	 */
86  	private int _clockSequence;
87  
88  	/**
89  	 * Last physical timestamp value <code>System.currentTimeMillis()</code>
90  	 * returned: used to catch (and report) cases where system clock goes
91  	 * backwards. Is also used to limit "drifting", that is, amount timestamps
92  	 * used can differ from the system time value. This value is not guaranteed
93  	 * to be monotonically increasing.
94  	 */
95  	private long _lastSystemTimestamp = 0L;
96  
97  	/**
98  	 * Timestamp value last used for generating a UUID (along with
99  	 * {@link #_clockCounter}. Usually the same as {@link #_lastSystemTimestamp}
100 	 * , but not always (system clock moved backwards). Note that this value is
101 	 * guaranteed to be monotonically increasing; that is, at given absolute
102 	 * time points t1 and t2 (where t2 is after t1), t1 <= t2 will always hold
103 	 * true.
104 	 */
105 	private long _lastUsedTimestamp = 0L;
106 
107 	/**
108 	 * First timestamp that can NOT be used without synchronizing using
109 	 * synchronization object ({@link #_syncer}). Only used when external
110 	 * timestamp synchronization (and persistence) is used, ie. when
111 	 * {@link #_syncer} is not null.
112 	 */
113 	private long _firstUnsafeTimestamp = Long.MAX_VALUE;
114 
115 	/**
116 	 * Counter used to compensate inadequate resolution of JDK system timer.
117 	 */
118 	private int _clockCounter = 0;

120 	public MutableUUIDTimer(Random rnd, TimestampSynchronizer sync) throws IOException {
121 		_random = rnd;
122 		_syncer = sync;
123 		initCounters(rnd);
124 		_lastSystemTimestamp = 0L;
125 		// This may get overwritten by the synchronizer
126 		_lastUsedTimestamp = 0L;
127 
128 		/*
129 		 * Ok, now; synchronizer can tell us what is the first timestamp value
130 		 * that definitely was NOT used by the previous incarnation. This can
131 		 * serve as the last used time stamp, assuming it is not less than value
132 		 * we are using now.
133 		 */
134 		if (sync != null) {
135 			long lastSaved = sync.initialize();
136 			if (lastSaved > _lastUsedTimestamp) {
137 				_lastUsedTimestamp = lastSaved;
138 			}
139 		}
140 
141 		/*
142 		 * Also, we need to make sure there are now no safe values (since
143 		 * synchronizer is not yet requested to allocate any):
144 		 */
145 		_firstUnsafeTimestamp = 0L; // ie. will always trigger sync.update()
146 	}

148 	private void initCounters(Random rnd) {
149 		/*
150 		 * Let's generate the clock sequence field now; as with counter, this
151 		 * reduces likelihood of collisions (as explained in UUID specs)
152 		 */
153 		_clockSequence = rnd.nextInt();
154 		/*
155 		 * Ok, let's also initialize the counter... Counter is used to make it
156 		 * slightly less likely that two instances of UUIDGenerator (from
157 		 * separate JVMs as no more than one can be created in one JVM) would
158 		 * produce colliding time-based UUIDs. The practice of using multiple
159 		 * generators, is strongly discouraged, of course, but just in case...
160 		 */
161 		_clockCounter = (_clockSequence >> 16) & 0xFF;
162 	}
163 
164 	public int getClockSequence() {
165 		return (_clockSequence & 0xFFFF);
166 	}

168 	public synchronized long getTimestamp(long systime) {
169 		/*
170 		 * Let's first verify that the system time is not going backwards;
171 		 * independent of whether we can use it:
172 		 */
173 		if (systime < _lastSystemTimestamp) {
174 			Logger.logWarning("System time going backwards! (got value "
175 					+ systime + ", last " + _lastSystemTimestamp);
176 			// Let's write it down, still
177 			_lastSystemTimestamp = systime;
178 		}
179 
180 		/*
181 		 * But even without it going backwards, it may be less than the last one
182 		 * used (when generating UUIDs fast with coarse clock resolution; or if
183 		 * clock has gone backwards over reboot etc).
184 		 */
185 		if (systime <= _lastUsedTimestamp) {
186 			/*
187 			 * Can we just use the last time stamp (ok if the counter hasn't hit
188 			 * max yet)
189 			 */
190 			if (_clockCounter < kClockMultiplier) { // yup, still have room
191 				systime = _lastUsedTimestamp;
192 			} else { // nope, have to roll over to next value and maybe wait
193 				long actDiff = _lastUsedTimestamp - systime;
194 				long origTime = systime;
195 				systime = _lastUsedTimestamp + 1L;
196 
197 				Logger.logWarning("Timestamp over-run: need to reinitialize random sequence");
198 
199 				/*
200 				 * Clock counter is now at exactly the multiplier; no use just
201 				 * anding its value. So, we better get some random numbers
202 				 * instead...
203 				 */
204 				initCounters(_random);
205 
206 				/*
207 				 * But do we also need to slow down? (to try to keep virtual
208 				 * time close to physical time; i.e. either catch up when system
209 				 * clock has been moved backwards, or when coarse clock
210 				 * resolution has forced us to advance virtual timer too far)
211 				 */
212 				if (actDiff >= kMaxClockAdvance) {
213 					slowDown(origTime, actDiff);
214 				}
215 			}
216 		} else {
217 			/*
218 			 * Clock has advanced normally; just need to make sure counter is
219 			 * reset to a low value (need not be 0; good to leave a small
220 			 * residual to further decrease collisions)
221 			 */
222 			_clockCounter &= 0xFF;
223 		}
224 
225 		_lastUsedTimestamp = systime;
226 
227 		/*
228 		 * Ok, we have consistent clock (virtual or physical) value that we can
229 		 * and should use. But do we need to check external syncing now?
230 		 */
231 		if (_syncer != null && systime >= _firstUnsafeTimestamp) {
232 			try {
233 				_firstUnsafeTimestamp = _syncer.update(systime);
234 			} catch (IOException ioe) {
235 				throw new RuntimeException("Failed to synchronize timestamp: "
236 						+ ioe);
237 			}
238 		}
239 
240 		/*
241 		 * Now, let's translate the timestamp to one UUID needs, 100ns unit
242 		 * offset from the beginning of Gregorian calendar...
243 		 */
244 		systime *= kClockMultiplierL;
245 		systime += kClockOffset;
246 
247 		// Plus add the clock counter:
248 		systime += _clockCounter;
249 		// and then increase
250 		++_clockCounter;
251 		return systime;
252 	}

254 	/**
255 	 * Method that constructs timestamp unique and suitable to use for
256 	 * constructing UUIDs. Default implementation just calls
257 	 * {@link #getTimestampSynchronized}, which is fully synchronized;
258 	 * sub-classes may choose to implemented alternate strategies
259 	 * 
260 	 * @return 64-bit timestamp to use for constructing UUID
261 	 */
262 	public final synchronized long getTimestamp() {
263 		return getTimestamp(System.currentTimeMillis());	
264 	}

272 	/*
273 	 * Method for accessing timestamp to use for creating UUIDs. Used ONLY by
274 	 * unit tests, hence protected.
275 	 */
276 	protected final void getAndSetTimestamp(byte[] uuidBytes) {
277 		long timestamp = getTimestamp();
278 
279 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_SEQUENCE] = (byte) _clockSequence;
280 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_SEQUENCE + 1] = (byte) (_clockSequence >> 8);
281 
282 		// Time fields aren't nicely split across the UUID, so can't just
283 		// linearly dump the stamp:
284 		int clockHi = (int) (timestamp >>> 32);
285 		int clockLo = (int) timestamp;
286 
287 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_HI] = (byte) (clockHi >>> 24);
288 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_HI + 1] = (byte) (clockHi >>> 16);
289 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_MID] = (byte) (clockHi >>> 8);
290 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_MID + 1] = (byte) clockHi;
291 
292 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_LO] = (byte) (clockLo >>> 24);
293 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_LO + 1] = (byte) (clockLo >>> 16);
294 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_LO + 2] = (byte) (clockLo >>> 8);
295 		uuidBytes[UUIDUtil.BYTE_OFFSET_CLOCK_LO + 3] = (byte) clockLo;
296 	}

304 	private final static int MAX_WAIT_COUNT = 50;
305 
306 	/**
307 	 * Simple utility method to use to wait for couple of milliseconds, to let
308 	 * system clock hopefully advance closer to the virtual timestamps used.
309 	 * Delay is kept to just a millisecond or two, to prevent excessive
310 	 * blocking; but that should be enough to eventually synchronize physical
311 	 * clock with virtual clock values used for UUIDs.
312 	 * 
313 	 * @param msecs
314 	 *            Number of milliseconds to wait for from current time point
315 	 */
316 	private final static void slowDown(long startTime, long actDiff) {
317 		/*
318 		 * First, let's determine how long we'd like to wait. This is based on
319 		 * how far ahead are we as of now.
320 		 */
321 		long ratio = actDiff / kMaxClockAdvance;
322 		long delay;
323 
324 		if (ratio < 2L) { // 200 msecs or less
325 			delay = 1L;
326 		} else if (ratio < 10L) { // 1 second or less
327 			delay = 2L;
328 		} else if (ratio < 600L) { // 1 minute or less
329 			delay = 3L;
330 		} else {
331 			delay = 5L;
332 		}
333 		Logger.logWarning("Need to wait for " + delay
334 				+ " milliseconds; virtual clock advanced too far in the future");
335 		long waitUntil = startTime + delay;
336 		int counter = 0;
337 		do {
338 			try {
339 				Thread.sleep(delay);
340 			} catch (InterruptedException ie) {
341 			}
342 			delay = 1L;
343 			/*
344 			 * This is just a sanity check: don't want an "infinite" loop if
345 			 * clock happened to be moved backwards by, say, an hour...
346 			 */
347 			if (++counter > MAX_WAIT_COUNT) {
348 				break;
349 			}
350 		} while (System.currentTimeMillis() < waitUntil);
351 	}
